SELECT YEAR(enrolment_date) as year, 
COUNT(*) AS total_students 
FROM students 
GROUP BY YEAR(enrolment_date);

SELECT office_address AS edificio, 
COUNT(*) tot_teachers 
FROM teachers 
GROUP BY office_address;

SELECT exam_id, 
AVG(vote) AS average_grade 
FROM exam_student 
GROUP BY exam_id;

SELECT department_id AS id_department,
COUNT(*) AS tot_corsi_laurea
FROM degrees
GROUP BY department_id;
